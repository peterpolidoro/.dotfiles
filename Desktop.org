This helper updates the live LXQt terminal integration while preserving the
rest of the generated LXQt settings files.  Keeping this as a script is less
brittle than checking a whole =globalkeyshortcuts.conf= or =panel.conf= snapshot
into Stow.

*.local/bin/use-kitty-lxqt:*

#+begin_src sh :tangle .local/bin/use-kitty-lxqt :shebang #!/bin/sh
set -eu

backup_once() {
    file=$1
    if [ -e "$file" ] && [ ! -e "$file.before-kitty" ]; then
        cp -a "$file" "$file.before-kitty"
    fi
}

ensure_key_value() {
    file=$1
    section=$2
    key=$3
    value=$4

    mkdir -p "$(dirname "$file")"
    case ":${prepared_files:-}:" in
        *":$file:"*) ;;
        *)
            if [ -e "$file" ]; then
                backup_once "$file"
            else
                : > "$file"
            fi
            prepared_files="${prepared_files:-}:$file"
            ;;
    esac

    awk -v section="$section" -v key="$key" -v value="$value" '
      BEGIN {
        in_section = 0
        seen_section = 0
        set_key = 0
      }
      $0 == "[" section "]" {
        if (seen_section && !set_key) {
          print key "=" value
        }
        in_section = 1
        seen_section = 1
        set_key = 0
        print
        next
      }
      /^\[/ {
        if (in_section && !set_key) {
          print key "=" value
          set_key = 1
        }
        in_section = 0
        print
        next
      }
      in_section && index($0, key "=") == 1 {
        print key "=" value
        set_key = 1
        next
      }
      { print }
      END {
        if (!seen_section) {
          print ""
          print "[" section "]"
          print key "=" value
        } else if (in_section && !set_key) {
          print key "=" value
        }
      }
    ' "$file" > "$file.tmp"
    mv "$file.tmp" "$file"
}

find_kitty_desktop() {
    for dir in \
        "$HOME/.guix-extra-profiles/desktop/desktop/share/applications" \
        "$HOME/.guix-home/profile/share/applications" \
        "$HOME/.guix-profile/share/applications" \
        /usr/share/applications
    do
        if [ -f "$dir/kitty.desktop" ]; then
            printf '%s\n' "$dir/kitty.desktop"
            return 0
        fi
    done
    printf 'kitty.desktop\n'
}

kitty_desktop=$(find_kitty_desktop)

qtxdg_file="$HOME/.config/lxqt-qtxdg.conf"
mkdir -p "$(dirname "$qtxdg_file")"
backup_once "$qtxdg_file"
printf 'TerminalEmulator=kitty.desktop\n' > "$qtxdg_file"

ensure_key_value \
    "$HOME/.config/pcmanfm-qt/lxqt/settings.conf" \
    "System" \
    "Terminal" \
    "kitty"

keys_file="$HOME/.config/lxqt/globalkeyshortcuts.conf"
if [ -e "$keys_file" ]; then
    backup_once "$keys_file"
    sed 's/^Comment=QTerminal$/Comment=Kitty/; s/^Exec=qterminal$/Exec=kitty/' \
        "$keys_file" > "$keys_file.tmp"
    mv "$keys_file.tmp" "$keys_file"
fi
ensure_key_value "$keys_file" "Control%2BAlt%2BT.1" "Comment" "Kitty"
ensure_key_value "$keys_file" "Control%2BAlt%2BT.1" "Enabled" "true"
ensure_key_value "$keys_file" "Control%2BAlt%2BT.1" "Exec" "kitty"
ensure_key_value "$keys_file" "Meta%2BReturn.42" "Comment" "Kitty"
ensure_key_value "$keys_file" "Meta%2BReturn.42" "Enabled" "true"
ensure_key_value "$keys_file" "Meta%2BReturn.42" "Exec" "kitty"

panel_file="$HOME/.config/lxqt/panel.conf"
if [ -e "$panel_file" ]; then
    backup_once "$panel_file"
    escaped_kitty_desktop=$(printf '%s\n' "$kitty_desktop" | sed 's/[\/&]/\\&/g')
    sed "s#/usr/share/applications/qterminal.desktop#$escaped_kitty_desktop#g; s#qterminal.desktop#$escaped_kitty_desktop#g" \
        "$panel_file" > "$panel_file.tmp"
    mv "$panel_file.tmp" "$panel_file"
fi

printf 'LXQt terminal settings now prefer kitty.\n'
printf 'Log out and back in, or restart lxqt-panel and pcmanfm-qt, for all changes to take effect.\n'
#+end_src

* Fonts and Themes

I use [[https://github.com/derat/xsettingsd][xsettingsd]] as a minimal settings daemon for Xorg applications.  It replaces similar daemons from desktop environments like GNOME and XFCE and enables me to use a simple configuration file like the following:

*.config/xsettingsd/xsettingsd.conf:*

#+begin_src conf :tangle .config/xsettingsd/xsettingsd.conf :noweb yes
Net/ThemeName "Matcha-dark-azul"
Net/IconThemeName "Papirus-Dark"
Gtk/DecorationLayout "menu:minimize,maximize,close"
Gtk/FontName "Cantarell 11"
Gtk/MonospaceFontName "Fira Mono 10"
Gtk/CursorThemeName "Adwaita"
Xft/Antialias 1
Xft/Hinting 0
Xft/HintStyle "hintnone"
#+end_src

I also have to do an extra step to make sure Emacs can find the font path from the "desktop" profile.

*.config/fontconfig/fonts.conf:*

#+begin_src xml :tangle .config/fontconfig/fonts.conf
<?xml version="1.0"?>
<!DOCTYPE fontconfig SYSTEM "fonts.dtd">
<fontconfig>
  <dir>~/.dotfiles/fonts</dir>
  <dir>~/.guix-extra-profiles/desktop/desktop/share/fonts</dir>
  <alias>
    <family>Apple Color Emoji</family>
    <prefer>
      <family>Noto Color Emoji</family>
    </prefer>
  </alias>
</fontconfig>
#+end_src
